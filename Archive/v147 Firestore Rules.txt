// v147 Firestore Rules — secure, simple, correct

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Require authentication for everything
    match /{document=**} {
      allow read, write: if request.auth != null;
    }

    // Foods collection
    match /Foods/{id} {
      allow read, write: if request.auth != null;
    }

    // Daily logs
    match /DailyLog/{day}/entries/{entry} {
      allow read, write: if request.auth != null;
    }
  }
}
