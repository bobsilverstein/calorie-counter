document.addEventListener("DOMContentLoaded", () => {
  const auth = firebase.auth();
  const db = firebase.firestore();

  const logView = document.getElementById("logView");
  const addFoodView = document.getElementById("addFoodView");
  const foodsView = document.getElementById("foodsView");
  const foodMgmtView = document.getElementById("foodMgmtView");
  const servingView = document.getElementById("servingView");

  const dateLabel = document.getElementById("dateLabel");
  const todayCaloriesEl = document.getElementById("todayCalories");

  const prevDayBtn = document.getElementById("prevDay");
  const nextDayBtn = document.getElementById("nextDay");

  const addFoodSearch = document.getElementById("addFoodSearch");
  const addFoodResults = document.getElementById("addFoodResults");

  const foodsSearch = document.getElementById("foodsSearch");
  const foodsList = document.getElementById("foodsList");

  const backFromAdd = document.getElementById("backFromAdd");
  const backFromFoods = document.getElementById("backFromFoods");
  const backFromFoodMgmt = document.getElementById("backFromFoodMgmt");
  const backFromServing = document.getElementById("backFromServing");

  const openFoodMgmt = document.getElementById("openFoodMgmt");

  const navLog = document.getElementById("navLog");
  const navFoods = document.getElementById("navFoods");
  const navAdd = document.getElementById("navAdd");

  const versionTag = document.getElementById("versionTag");

  const servingFoodName = document.getElementById("servingFoodName");
  const servingCount = document.getElementById("servingCount");
  const applyServing = document.getElementById("applyServing");

  let currentDate = new Date();
  let submitLock = false;
  let selectedServingFood = null;

  function fmtDate(d) {
    return d.toISOString().split("T")[0];
  }

  // VERSION NUMBER — EXACT SPEC
  function updateVersionTag() {
    const now = new Date();
    const mm = String(now.getMonth() + 1).padStart(2, "0");
    const dd = String(now.getDate()).padStart(2, "0");
    const yyyy = now.getFullYear();
    const time = now.toLocaleTimeString([], { hour:"2-digit", minute:"2-digit" });

    versionTag.textContent =
      `Version 109 (increment automatically on each save), ${mm}/${dd}/${yyyy} ${time}`;
  }

  updateVersionTag(); // initial load

  function show(view) {
    [logView, addFoodView, foodsView, foodMgmtView, servingView]
      .forEach(v => v.classList.remove("active"));
    view.classList.add("active");
  }

  auth.onAuthStateChanged(user => {
    if (!user) {
      logView.innerHTML = "<p style='padding:12px;'>Please sign in.</p>";
      todayCaloriesEl.textContent = "0";
      return;
    }
    renderLog();
  });

  function renderLog() {
    dateLabel.textContent = currentDate.toDateString();

    const uid = auth.currentUser.uid;
    const dateKey = fmtDate(currentDate);

    db.collection("logs").doc(uid).collection("days").doc(dateKey).get()
      .then(doc => {
        const data = doc.exists ? doc.data() : { meals:{} };
        const meals = data.meals || {};

        let total = 0;
        Object.values(meals).forEach(arr => arr.forEach(i => total += i.calories));
        todayCaloriesEl.textContent = total;

        logView.innerHTML = buildMeals(meals);
      });
  }

  function buildMeals(meals) {
    const order = ["Breakfast","Lunch","Dinner","Snacks"];
    let html = "";

    order.forEach(meal => {
      const items = meals[meal] || [];
      html += `
        <div class="meal-bar">${meal}</div>
        <div class="meal-items">
          ${items.map(i => `<div class="meal-item">${i.name} — ${i.calories} cal</div>`).join("")}
        </div>
      `;
    });

    return html;
  }

  prevDayBtn.onclick = () => {
    currentDate.setDate(currentDate.getDate() - 1);
    renderLog();
  };

  nextDayBtn.onclick = () => {
    const today = new Date(); today.setHours(0,0,0,0);
    const next = new Date(currentDate); next.setDate(next.getDate() + 1);
    if (next <= today) { currentDate = next; renderLog(); }
  };

  navLog.onclick = () => show(logView);
  navFoods.onclick = () => { show(foodsView); loadFoods(); };
  navAdd.onclick = () => { show(addFoodView); addFoodSearch.value=""; addFoodResults.innerHTML=""; };

  addFoodSearch.oninput = () => {
    const q = addFoodSearch.value.trim();
    if (!q) { addFoodResults.innerHTML=""; return; }

    db.collection("foods")
      .orderBy("name").startAt(q).endAt(q+"\uf8ff").limit(20)
      .get()
      .then(snapshot => {
        addFoodResults.innerHTML = snapshot.docs.map(doc => {
          const f = doc.data();
          return `<div class="food-result" data-id="${doc.id}">
            ${f.name} — ${f.calories} cal
          </div>`;
        }).join("");

        [...addFoodResults.children].forEach(el => {
          el.onclick = () => openServingForFood(el.dataset.id);
        });
      });
  };

  function openServingForFood(id) {
    db.collection("foods").doc(id).get().then(doc => {
      if (!doc.exists) return;
      const f = doc.data();
      selectedServingFood = { id, name:f.name, calories:f.calories };
      servingFoodName.textContent = `${f.name} — ${f.calories} cal per serving`;
      servingCount.value = "1";
      show(servingView);
    });
  }

  applyServing.onclick = () => {
    if (!selectedServingFood || submitLock) return;
    submitLock = true;

    const uid = auth.currentUser.uid;
    const dateKey = fmtDate(currentDate);
    const servings = parseFloat(servingCount.value || "1");
    const totalCals = Math.round(selectedServingFood.calories * servings);

    const dayRef = db.collection("logs").doc(uid).collection("days").doc(dateKey);

    dayRef.set({
      meals:{
        Snacks: firebase.firestore.FieldValue.arrayUnion({
          name:`${selectedServingFood.name} x${servings}`,
          calories: totalCals
        })
      }
    }, { merge:true })
    .then(() => {
      submitLock = false;
      updateVersionTag();   // ← EXACTLY AS YOU SPECIFIED
      show(logView);
      renderLog();
    });
  };

  function loadFoods() {
    const q = foodsSearch.value.trim();

    db.collection("foods")
      .orderBy("name").startAt(q).endAt(q+"\uf8ff").limit(50)
      .get()
      .then(snapshot => {
        foodsList.innerHTML = snapshot.docs.map(doc => {
          const f = doc.data();
          return `<div class="food-result" data-id="${doc.id}">
            ${f.name} — ${f.calories} cal
          </div>`;
        }).join("");

        [...foodsList.children].forEach(el => {
          el.onclick = () => openServingForFood(el.dataset.id);
        });
      });
  }

  foodsSearch.oninput = loadFoods;

  openFoodMgmt.onclick = () => show(foodMgmtView);

  backFromAdd.onclick = () => show(logView);
  backFromFoods.onclick = () => show(logView);
  backFromFoodMgmt.onclick = () => show(foodsView);
  backFromServing.onclick = () => show(addFoodView);
});
